# Measure Categorization Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Reorganize 807 measures across 23 tables into subject-area tables with hierarchical display folders, following the spec in `docs/superpowers/specs/2026-05-07-measure-categorization-design.md`.

**Architecture:** Two-track approach — (1) `pbi measure update --display-folder` for tables staying in place (UK, US, UAE, ROW, Inventory, Retail KPIs, FootFall, etc.); (2) TMDL export → Python reorganization script → TMDL import for the 317-measure `Key Measures` split into 5 new tables. One qualified DAX reference must be fixed before the move.

**Tech Stack:** pbi-cli, PowerShell, Python 3, TMDL format

# ── Helper: find measure files in TMDL export ─────────────────────────────────
def find_measure_files(tmdl_dir, table_name):
    """Return dict of {measure_name: file_path} for a table."""
    result = {}
    tables_dir = os.path.join(tmdl_dir, "tables")
    if not os.path.exists(tables_dir):
        # Try single-file TMDL format
        print(f"WARNING: {tables_dir} not found. Inspect export structure manually.")
        return result
    for entry in os.scandir(tables_dir):
        if entry.is_dir() and entry.name.strip("'") == table_name:
            measures_dir = os.path.join(entry.path, "measures")
            if os.path.exists(measures_dir):
                for f in os.scandir(measures_dir):
                    if f.name.endswith(".tmdl"):
                        mname = f.name[:-5].strip("'")
                        result[mname] = f.path
            break
    return result

def update_display_folder(content, new_folder):
    """Replace or insert displayFolder in a TMDL measure block."""
    folder_val = new_folder.replace("\\", "\\\\")
    if re.search(r'displayFolder:', content):
        return re.sub(r'displayFolder:\s*"[^"]*"', f'displayFolder: "{folder_val}"', content)
    # Insert after first line
    lines = content.splitlines(keepends=True)
    insert_at = 1
    for i, line in enumerate(lines[1:], 1):
        if line.strip() and not line.strip().startswith("//"):
            insert_at = i
            break
    lines.insert(insert_at, f'\tdisplayFolder: "{folder_val}"\n')
    return "".join(lines)

def ensure_table_dir(tmdl_dir, table_name):
    """Create table directory and placeholder tmdl if needed."""
    tables_dir = os.path.join(tmdl_dir, "tables")
    table_dir = os.path.join(tables_dir, f"'{table_name}'")
    measures_dir = os.path.join(table_dir, "measures")
    os.makedirs(measures_dir, exist_ok=True)
    table_file = os.path.join(table_dir, f"'{table_name}'.tmdl")
    if not os.path.exists(table_file):
        with open(table_file, "w", encoding="utf-8") as f:
            f.write(f"table '{table_name}'\n\tisHidden\n\n")
    return measures_dir

# ── Main ──────────────────────────────────────────────────────────────────────
if __name__ == "__main__":
    km_files = find_measure_files(TMDL_DIR, "Key Measures")
    if not km_files:
        print("ERROR: Could not find Key Measures files. Check TMDL structure.")
        print("Directory listing:")
        for root, dirs, files in os.walk(TMDL_DIR):
            level = root.replace(TMDL_DIR, '').count(os.sep)
            indent = ' ' * 2 * level
            print(f'{indent}{os.path.basename(root)}/')
            if level < 3:
                for f in files:
                    print(f'{indent}  {f}')
        exit(1)

    moved, skipped = 0, []
    new_tables = set(t for t, _ in MAPPING.values())

    for measure_name, (new_table, display_folder) in MAPPING.items():
        if measure_name not in km_files:
            skipped.append(measure_name)
            continue
        src = km_files[measure_name]
        with open(src, "r", encoding="utf-8") as f:
            content = f.read()
        content = update_display_folder(content, display_folder)
        dest_dir = ensure_table_dir(TMDL_DIR, new_table)
        dest = os.path.join(dest_dir, os.path.basename(src))
        with open(dest, "w", encoding="utf-8") as f:
            f.write(content)
        os.remove(src)
        moved += 1

    # Report unmapped measures remaining in Key Measures
    remaining = find_measure_files(TMDL_DIR, "Key Measures")
    print(f"\nMoved: {moved} measures")
    print(f"Skipped (not found in export): {skipped}")
    print(f"Remaining in Key Measures ({len(remaining)}): {list(remaining.keys())}")

## Notes

- **Mapping dict duplicate keys** — several measures (e.g., `MTD Dilution%`, `MTD Full Price Sales`, `YTD Full Sales`, `LYTD Full Sales`, `LY MTD DILUTION%`) appear in the MAPPING dict twice with different targets. Python resolves this by keeping the **last** assignment, which is intentional — the final assignment is always the correct one per the spec. The earlier entries are harmless overrides left in place for readability context. Do not remove them.
- **Duplicate measures** (`Forcast Sales`/`Budgeted Sales`, `GP`/`GP-1`, `CGS`/`CGS-1`, `Net Sales`/`Net Sales-1`, `Sales Tax`/`Sales Tax-1`, `LY GP`/`LY G.P`, `MTD TGT GP Var%`/`MTD TGT GP VAR`) — these are not resolved in this plan.
